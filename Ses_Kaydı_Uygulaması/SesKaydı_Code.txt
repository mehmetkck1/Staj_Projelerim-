  var formatDuration = function(seconds) {
    if (!seconds || isNaN(seconds)) return '0:00';
    var s = Math.floor(seconds);
    var h = Math.floor(s / 3600);
    var m = Math.floor((s % 3600) / 60);
    var sec = s % 60;
    if (h > 0) return h + ':' + (m < 10 ? '0' : '') + m + ':' + (sec < 10 ? '0' : '') + sec;
    return m + ':' + (sec < 10 ? '0' : '') + sec;
  };
  var formatFileSize = function(bytes) {
    if (!bytes) return '0 KB';
    if (bytes < 1024) return bytes + ' B';
    if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(1) + ' KB';
    return (bytes / (1024 * 1024)).toFixed(1) + ' MB';
  };
  var formatDate = function(dateStr) {
    if (!dateStr) return '';
    var d = new Date(dateStr);
    var now = new Date();
    var diff = now - d;
    if (diff < 60000) return 'Just now';
    if (diff < 3600000) return Math.floor(diff / 60000) + 'm ago';
    if (diff < 86400000) return Math.floor(diff / 3600000) + 'h ago';
    if (diff < 604800000) return Math.floor(diff / 86400000) + 'd ago';
    return d.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
  };

var PlayerModal = function(props) {
    var visible = props.visible;
    var recording = props.recording;
    var onClose = props.onClose;

    var nativePause = audioHook.pause;
    var nativeResume = audioHook.resume;
    var nativeStop = audioHook.stop;
    var nativeIsPlaying = audioHook.isPlaying;
    var nativeCurrentTime = audioHook.currentTime;
    var nativeDuration = audioHook.duration;
    var nativeSeekTo = audioHook.seekTo;
    var webAudioRef = useRef(null);
    var webIsPlayingState = useState(false);
    var webIsPlaying = webIsPlayingState[0];
    var setWebIsPlaying = webIsPlayingState[1];
    var webCurrentTimeState = useState(0);
    var webCurrentTime = webCurrentTimeState[0];
    var setWebCurrentTime = webCurrentTimeState[1];
    var webDurationState = useState(0);
    var webDuration = webDurationState[0];
    var setWebDuration = webDurationState[1];
    var speedState = useState(1);
    var speed = speedState[0];
    var setSpeed = speedState[1];
    var isCurrentlyPlaying = Platform.OS === 'web' ? webIsPlaying : nativeIsPlaying;
    var displayCurrent = Platform.OS === 'web' ? webCurrentTime : nativeCurrentTime;
    var displayDuration = Platform.OS === 'web' ? webDuration : nativeDuration;
    useEffect(function() {
      if (!visible) {
        if (Platform.OS === 'web') {
          if (webAudioRef.current) { webAudioRef.current.pause(); webAudioRef.current = null; }
          setWebIsPlaying(false); setWebCurrentTime(0); setWebDuration(0);
        } else { nativeStop(); }
      }
    }, [visible]);
    useEffect(function() {
      return function() {
        if (Platform.OS === 'web' && webAudioRef.current) { webAudioRef.current.pause(); webAudioRef.current = null; }
        else if (Platform.OS !== 'web') { nativeStop(); }
      };
    }, []);
    var handlePlayPause = function() {
      if (!recording || !recording.audio_file_path) return;
      if (Platform.OS === 'web') {
        if (!webAudioRef.current) {
          webAudioRef.current = new Audio(recording.audio_file_path);
          webAudioRef.current.onloadedmetadata = function() { setWebDuration(webAudioRef.current.duration || 0); };
          webAudioRef.current.ontimeupdate = function() { setWebCurrentTime(webAudioRef.current.currentTime || 0); };
          webAudioRef.current.onended = function() { setWebIsPlaying(false); setWebCurrentTime(0); };
          webAudioRef.current.playbackRate = speed;
        }
        if (webIsPlaying) { webAudioRef.current.pause(); setWebIsPlaying(false); }
        else { webAudioRef.current.play(); setWebIsPlaying(true); }
      } else {
        if (nativeIsPlaying) { nativePause(); } else { nativePlay(recording.audio_file_path); }
      }
    };
    var handleSeek = function(ratio) {
      var dur = displayDuration || (recording ? recording.duration_seconds : 0);
      var newTime = Math.max(0, Math.min(1, ratio)) * dur;
      if (Platform.OS === 'web') {
        if (webAudioRef.current) { webAudioRef.current.currentTime = newTime; setWebCurrentTime(newTime); }
      } else { nativeSeekTo(newTime); }
    };
    var handleRewind = function() {
      var newTime = Math.max(0, displayCurrent - 10);
      if (Platform.OS === 'web') { if (webAudioRef.current) { webAudioRef.current.currentTime = newTime; setWebCurrentTime(newTime); } }
      else { nativeSeekTo(newTime); }
    };
    var handleForward = function() {
      var dur = displayDuration || (recording ? recording.duration_seconds : 0);
      var newTime = Math.min(dur, displayCurrent + 10);
      if (Platform.OS === 'web') { if (webAudioRef.current) { webAudioRef.current.currentTime = newTime; setWebCurrentTime(newTime); } }
      else { nativeSeekTo(newTime); }
    };
    var handleSpeedChange = function() {
      var speeds = [0.5, 1, 1.5, 2];
      var idx = speeds.indexOf(speed);
      var next = speeds[(idx + 1) % speeds.length];
      setSpeed(next);
      if (Platform.OS === 'web' && webAudioRef.current) { webAudioRef.current.playbackRate = next; }
    };
    var dur = (displayDuration > 0 ? displayDuration : (recording ? recording.duration_seconds : 0)) || 1;
    var progress = Math.min(1, displayCurrent / dur);

    return React.createElement(Modal, { visible: visible, animationType: 'slide', transparent: true, onRequestClose: onClose },

          React.createElement(TouchableOpacity, { onPress: onClose, style: { position: 'absolute', top: 20, right: 20, padding: 8 }, componentId: 'player-close-btn' },
            React.createElement(MaterialIcons, { name: 'close', size: 22, color: theme.colors.textSecondary })
          ),
          React.createElement(View, { style: { width: 64, height: 64, borderRadius: 20, backgroundColor: theme.colors.primary, alignItems: 'center', justifyContent: 'center', alignSelf: 'center', marginBottom: 16 } },

          ),
          React.createElement(Text, { style: { fontSize: 18, fontWeight: '700', color: theme.colors.textPrimary, textAlign: 'center', marginBottom: 4 }, numberOfLines: 1 }, recording.title || 'Kayıt'),

          React.createElement(View, { style: { flexDirection: 'row', alignItems: 'flex-end', justifyContent: 'center', height: 60, gap: 2, marginBottom: 12 }, componentId: 'player-waveform' },
            Array.from({ length: 40 }).map(function(_, i) {
              var h = 8 + Math.sin(i * 0.7) * 14;
              var filled = (i / 40) < progress;
              return React.createElement(View, { key: String(i), style: { width: 4, height: Math.max(4, h), borderRadius: 2, backgroundColor: filled ? theme.colors.primary : theme.colors.border } });

    var recordings = recordingsQuery.data || [];
    var insertBackupMut = useMutation('backups', 'insert');
    var deleteBackupMut = useMutation('backups', 'delete');
    var filePickerHook = useFilePicker();
    var pickDocument = filePickerHook.pickDocument;

    var shareHook = useShare();

    var creatingState = useState(false);
    var creating = creatingState[0];
    var setCreating = creatingState[1];
    var totalSize = useMemo(function() { return recordings.reduce(function(acc, r) { return acc + (r.file_size_bytes || 0); }, 0); }, [recordings]);
    var totalDuration = useMemo(function() { return recordings.reduce(function(acc, r) { return acc + (r.duration_seconds || 0); }, 0); }, [recordings]);
    var handleCreateBackup = function() {
      setCreating(true);
      var ts = new Date();
      var name = 'VoiceVault_Yedek_' + ts.toISOString().split('T')[0];
      insertBackupMut.mutate({ backup_name: name, recording_count: recordings.length, total_size_bytes: totalSize, backup_file_path: null, backup_timestamp: ts.toISOString(), backup_type: 'manual' }).then(function() {
        setCreating(false);
        backupsQuery.refetch();
        if (Platform.OS === 'web') window.alert(TR.backupCreatedSuccess + ' ' + name);
        else Alert.alert(TR.backupCreated, name + ' ' + TR.backupCreatedSuccess);
      }).catch(function(err) {
        setCreating(false);
        if (Platform.OS === 'web') window.alert(err.message); else Alert.alert(TR.error, err.message);
      });
    };
    var handleImportBackup = function() {
      pickDocument({ type: ['application/json', 'application/zip', '*/*'] }).then(function(result) {
        if (!result.cancelled && result.file) {
          insertBackupMut.mutate({ backup_name: result.file.name, recording_count: 0, total_size_bytes: result.file.size || 0, backup_file_path: result.file.uri, backup_timestamp: new Date().toISOString(), backup_type: 'imported' }).then(function() {
            backupsQuery.refetch();
            if (Platform.OS === 'web') window.alert(TR.backupImportedSuccess + ' ' + result.file.name);
            else Alert.alert(TR.backupImported, result.file.name + ' ' + TR.backupImportedSuccess);
          });
        }
      });
    };
    var handleDeleteBackup = function(backup) {
      var doDelete = function() { deleteBackupMut.mutate({ id: backup.id }).then(function() { backupsQuery.refetch(); }); };
      if (Platform.OS === 'web') { if (window.confirm('"' + backup.backup_name + '" ' + TR.deleteQuestion + '?')) doDelete(); }
      else Alert.alert(TR.deleteBackupQuestion, '"' + backup.backup_name + '" ' + TR.deleteQuestion + '?', [{ text: TR.cancel, style: 'cancel' }, { text: TR.delete, style: 'destructive', onPress: doDelete }]);
    };
    var handleShareBackup = function(backup) {
      share({ message: 'VoiceVault ' + TR.backup + ': ' + backup.backup_name + '\n' + TR.recordings + ': ' + backup.recording_count + '\nTarih: ' + formatDate(backup.backup_timestamp) }).then(function(r) { if (r.error) { if (Platform.OS !== 'web') Alert.alert(TR.error, r.error); } });
    };

          React.createElement(View, { style: { flexDirection: 'row', gap: 12, marginBottom: 20 } },

var RecordingOverlayState = {
  isVisible: false,
  setIsVisible: null,
  isRecording: false,
  setIsRecording: null,
  duration: 0,
  setDuration: null,
  isRecordingPaused: false,
  setIsRecordingPaused: null,
  waveform: [],
  setWaveform: null,
  lastRecUri: null,
  setLastRecUri: null,
  showSave: false,
  setShowSave: null
};

  // @section:RecordingOverlay @depends:[imports,theme,useTheme,SaveRecordingModal,helpers]
var RecordingOverlay = function(props) {
  var visible = props.visible;
  var onClose = props.onClose;
  var themeCtx = useTheme();
  var theme = themeCtx.theme;
  var insets = useSafeAreaInsets();
  
  const audioHook = useAudio();
  var startRec = audioHook.startRecording;
  var stopRec = audioHook.stopRecording;
  var isRecording = audioHook.isRecording;
  
  var isRecordingPausedState = useState(false);
  var isRecordingPaused = isRecordingPausedState[0];
  var setIsRecordingPaused = isRecordingPausedState[1];
  
  var durationState = useState(0);
  var duration = durationState[0];
  var setDuration = durationState[1];
  
  var timerRef = useRef(null);
  
  var waveformState = useState(Array.from({ length: 24 }, function() { return 8; }));
  var waveform = waveformState[0];
  var setWaveform = waveformState[1];
  
  var showSaveState = useState(false);
  var showSave = showSaveState[0];
  var setShowSave = showSaveState[1];
  
  var lastRecUriState = useState(null);
  var lastRecUri = lastRecUriState[0];
  var setLastRecUri = lastRecUriState[1];
  
  var insertRec = useMutation('recordings', 'insert').mutate;
  var foldersQuery = useQuery('folders', {}, { column: 'name', ascending: true });
  var folders = foldersQuery.data || [];
  
  useEffect(function() {
    if (isRecording && !isRecordingPaused) {
      timerRef.current = setInterval(function() {
        setDuration(function(prev) { return prev + 1; });
        setWaveform(function() { return Array.from({ length: 24 }, function() { return 6 + Math.random() * 34; }); });
      }, 1000);
    } else {
      if (timerRef.current) { clearInterval(timerRef.current); timerRef.current = null; }
      if (!isRecording) { setWaveform(Array.from({ length: 24 }, function() { return 8; })); }
    }
    return function() { if (timerRef.current) { clearInterval(timerRef.current); timerRef.current = null; } };
  }, [isRecording, isRecordingPaused]);
  
  useEffect(function() {
    if (!visible) {
      if (timerRef.current) { clearInterval(timerRef.current); timerRef.current = null; }
      setDuration(0);
      setIsRecordingPaused(false);
      setWaveform(Array.from({ length: 24 }, function() { return 8; }));
    }
  }, [visible]);
  
  var handleStartRecording = function() {
    setDuration(0);
    startRec().catch(function(err) {
      if (Platform.OS === 'web') window.alert(TR.microphoneError + err.message);
      else Alert.alert(TR.error, err.message);
    });
  };
  
  var handleStopRecording = function() {
    stopRec().then(function(result) {
      if (result && result.uri) { setLastRecUri(result.uri); setShowSave(true); }
      else if (result && result.error) {
        if (Platform.OS === 'web') window.alert(result.error); else Alert.alert(TR.error, result.error);
      }
      setIsRecordingPaused(false);
    });
  };
  
  var handleSaveRecording = function(title, folderId) {
    insertRec({ title: title, duration_seconds: duration, file_size_bytes: Math.floor(duration * 16000), folder_id: folderId || null, is_favorite: false, audio_file_path: lastRecUri || 'local://recording_' + Date.now(), recording_date: new Date().toISOString() }).then(function() {
      setShowSave(false); setLastRecUri(null); setDuration(0);
      onClose();
      if (Platform.OS === 'web') window.alert(TR.recordingSaved); else Alert.alert(TR.ok, TR.recordingSaved);
    }).catch(function(err) {
      if (Platform.OS === 'web') window.alert(err.message); else Alert.alert(TR.error, err.message);
    });
  };
  
  var handleDiscardRecording = function() {
    if (Platform.OS === 'web') {
      if (window.confirm(TR.discardRecording + '?')) { setShowSave(false); setLastRecUri(null); setDuration(0); }
    } else {
      Alert.alert(TR.discard, TR.discardRecording + '?', [{ text: TR.keep, style: 'cancel' }, { text: TR.discard, style: 'destructive', onPress: function() { setShowSave(false); setLastRecUri(null); setDuration(0); } }]);
    }
  };
  
  if (!visible) return null;
  
  return React.createElement(View, { style: { position: 'absolute', top: 0, left: 0, right: 0, bottom: 0, backgroundColor: 'rgba(0,0,0,0.5)', zIndex: 999, justifyContent: 'flex-end' } },
    React.createElement(View, { style: { backgroundColor: theme.colors.background, borderTopLeftRadius: 32, borderTopRightRadius: 32, paddingHorizontal: 24, paddingTop: 24, paddingBottom: insets.bottom + 24, minHeight: '60%' } },
      React.createElement(View, { style: { width: 40, height: 4, backgroundColor: theme.colors.border, borderRadius: 2, alignSelf: 'center', marginBottom: 24 } }),
      
      React.createElement(View, { style: { alignItems: 'center', marginBottom: 40 } },
        React.createElement(Text, { style: { fontSize: 56, fontWeight: '100', color: theme.colors.textPrimary, letterSpacing: 2, fontVariant: ['tabular-nums'] }, componentId: 'overlay-timer' }, formatDuration(duration))
      ),
      
      React.createElement(View, { style: { flexDirection: 'row', alignItems: 'flex-end', justifyContent: 'center', height: 60, gap: 3, marginBottom: 48 }, componentId: 'overlay-waveform' },
        waveform.map(function(h, i) {
          var active = isRecording && !isRecordingPaused;
          return React.createElement(View, { key: String(i), style: { width: 6, height: active ? h : 8, borderRadius: 3, backgroundColor: active ? (i % 3 === 0 ? theme.colors.primary : theme.colors.accent) : theme.colors.border } });
        })
      ),
      
      React.createElement(View, { style: { flexDirection: 'row', alignItems: 'center', justifyContent: 'center', gap: 24, marginBottom: 32 } },
        isRecording ? React.createElement(TouchableOpacity, { onPress: function() { setIsRecordingPaused(function(p) { return !p; }); }, style: { width: 52, height: 52, borderRadius: 26, backgroundColor: theme.colors.cardAlt, alignItems: 'center', justifyContent: 'center', borderWidth: 1.5, borderColor: theme.colors.border }, componentId: 'overlay-pause-resume-btn' },
          React.createElement(MaterialIcons, { name: isRecordingPaused ? 'play-arrow' : 'pause', size: 24, color: theme.colors.primary })
        ) : React.createElement(View, { style: { width: 52, height: 52 } }),
        
        React.createElement(TouchableOpacity, { onPress: isRecording ? handleStopRecording : handleStartRecording, style: { width: 80, height: 80, borderRadius: 40, backgroundColor: isRecording ? theme.colors.error : theme.colors.primary, alignItems: 'center', justifyContent: 'center', shadowColor: isRecording ? theme.colors.error : theme.colors.primary, shadowOpacity: 0.4, shadowRadius: 16, elevation: 10 }, componentId: 'overlay-main-record-btn' },
          React.createElement(MaterialIcons, { name: isRecording ? 'stop' : 'mic', size: 36, color: '#FFFFFF' })
        ),
        
        React.createElement(TouchableOpacity, { onPress: onClose, style: { width: 52, height: 52, borderRadius: 26, backgroundColor: theme.colors.cardAlt, alignItems: 'center', justifyContent: 'center', borderWidth: 1.5, borderColor: theme.colors.border }, componentId: 'overlay-close-btn' },
          React.createElement(MaterialIcons, { name: 'close', size: 24, color: theme.colors.textSecondary })
        )
      ),
      
      isRecording ? React.createElement(View, { style: { flexDirection: 'row', alignItems: 'center', justifyContent: 'center', gap: 6 } },
        React.createElement(View, { style: { width: 8, height: 8, borderRadius: 4, backgroundColor: theme.colors.error } }),
        React.createElement(Text, { style: { fontSize: 12, color: theme.colors.error, fontWeight: '600' } }, isRecordingPaused ? TR.paused : TR.recording)
      ) : null
    ),
    React.createElement(SaveRecordingModal, { visible: showSave, onSave: handleSaveRecording, onDiscard: handleDiscardRecording, duration: duration, folders: folders, insetsBottom: insets.bottom })
  );
};
